// sum of squares of n integers
	int n, sum=0, count=0;
	printf("Enter an integer",n);
	scanf("%d",&n);
	while (count<n) {
		int i;
		printf("Enter an integer",i);
		scanf("%d", &i);
		sum=((i*i)+sum);
		count=count+1;
	}
	printf("The sum of squares is: %d\n", sum);
	return=0;
}
